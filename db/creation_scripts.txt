CREATE TABLE users (
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	user_id TEXT(50) NOT NULL UNIQUE,
	user_name TEXT(50) NOT NULL,
	date_request DATETIME NOT NULL
);

CREATE TABLE user_requests (
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	user_id TEXT(50) NOT NULL,
	date_request DATETIME NOT NULL,
	city_request TEXT(50) NOT NULL,
	command_request TEXT(20) NOT NULL,
	hotels TEXT(255) NOT NULL,
	CONSTRAINT user_requests_FK FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE
);
